  <script>
    // Registrierung des Service Workers
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.register('service-worker.js')
      .then(function(registration) {
        console.log('Service Worker registriert mit Scope:', registration.scope);
      })
      .catch(function(error) {
        console.log('Service Worker Registrierung fehlgeschlagen:', error);
      });
    }
    
    // Funktion, um die Berechtigung f端r Notifications anzufordern
    function askNotificationPermission() {
      if (!('Notification' in window)) {
        alert('Dieser Browser unterst端tzt keine Notifications.');
      } else {
        Notification.requestPermission().then(function(permission) {
          if (permission === 'granted') {
            console.log('Notification permission granted.');
          }
        });
      }
    }
    
    // Test-Button f端r eine Notification
    document.getElementById('notifyBtn').addEventListener('click', function() {
      if (Notification.permission === 'granted') {
        navigator.serviceWorker.getRegistration().then(function(reg) {
          reg.showNotification('Tiny Habits', {
            body: 'Dies ist eine Test-Benachrichtigung!',
            icon: 'icon-192.png'
          });
        });
      } else {
        askNotificationPermission();
      }
    });
    
    // Optionale Installationserinnerung (f端r den Home-Bildschirm)
    window.addEventListener('beforeinstallprompt', (e) => {
      e.preventDefault();
      console.log('App kann installiert werden!');
    });
  </script>
